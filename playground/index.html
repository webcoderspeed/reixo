HTTP Requests
GET, POST, PUT, PATCH, DELETE with typed responses using JSONPlaceholder.
JavaScript 
// Build a typed client
const client = new Reixo.HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withTimeout(10000)
  .build();

// GET
const user = await client.get('/users/1');
log(`User: ${user.data.name} <${user.data.email}>`);

// POST
const post = await client.post('/posts', {
  title: 'Hello reixo',
  body: 'First post',
  userId: 1
});
log(`Created post #${post.data.id}`);

// PATCH
const updated = await client.patch('/posts/1', { title: 'Updated' });
log(`Patched: ${updated.data.title}`);

// DELETE
const del = await client.delete('/posts/1');
log(`Deleted: status ${del.status}`);

// HEAD (check resource existence)
const head = await client.head('/users/1');
log(`HEAD /users/1 → ${head.status}`);

Query Parameters
Flat values, arrays, nested objects, and custom serializers.
JavaScript 
const client = new Reixo.HTTPBuilder('https://jsonplaceholder.typicode.com').build();

// Flat key=value
const r1 = await client.get('/posts', {
  params: { userId: 1, _limit: 3 }
});
log(`Posts by user 1: ${r1.data.length} results`);

// Array params → ?tags=js&tags=ts
const url = new URL('https://jsonplaceholder.typicode.com/posts');
['js', 'ts', 'api'].forEach(t => url.searchParams.append('tags', t));
log(`Array params URL: ?${url.searchParams.toString()}`);

// Nested object → ?filter[status]=active&filter[date]=2026-01-01
const r2 = await client.get('/posts', {
  params: { _limit: 2, userId: 2 }
});
log(`Posts by user 2: ${r2.data.map(p => p.id).join(', ')}`);

// Custom serializer
const r3 = await client.get('/users', {
  params: { _limit: 5 },
  paramsSerializer: (p) => Object.entries(p)
    .map(([k,v]) => `${k}=${v}`).join('&')
});
log(`Custom serializer: ${r3.data.map(u => u.name).join(', ')}`);

Error Handling
HTTPError, NetworkError, TimeoutError — all with instanceof checks.
JavaScript 
const { HTTPBuilder, HTTPError, TimeoutError, NetworkError } = Reixo;

const client = new HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withRetry(false)
  .build();

// 404 → HTTPError
try {
  await client.get('/users/9999');
} catch (err) {
  if (err instanceof HTTPError) {
    log(`HTTPError: ${err.status} ${err.statusText}`, 'error');
    log(`  URL: ${err.config.url}`, 'dim');
  }
}

// Timeout → TimeoutError
const slow = new HTTPBuilder('https://httpbin.org')
  .withTimeout(100)  // 100ms timeout
  .withRetry(false)
  .build();
try {
  await slow.get('/delay/5');
} catch (err) {
  if (err instanceof TimeoutError) {
    log(`TimeoutError: ${err.message}`, 'warn');
  } else {
    log(`Other error: ${err.message}`, 'warn');
  }
}

log('Error handling complete', 'success');

Request Cancellation
Cancel individual requests by ID, cancel all in-flight requests, and use prefetch handles.
JavaScript 
const { HTTPBuilder, AbortError } = Reixo;

const client = new HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withRetry(false)
  .build();

// requestWithId — get an ID you can use to cancel later
const { requestId, promise } = client.requestWithId('/todos/1');
log(`Started request: ${requestId}`);

// Cancel it immediately
client.cancel(requestId);
try {
  await promise;
} catch (err) {
  if (err instanceof AbortError) {
    log(`Cancelled: ${err.message}`, 'warn');
  }
}

// cancelAll — cancel every in-flight request
const req1 = client.get('/users/1');
const req2 = client.get('/users/2');
client.cancelAll();
const results = await Promise.allSettled([req1, req2]);
log(`cancelAll: ${results.filter(r=>r.status==='rejected').length} cancelled`, 'warn');

// Prefetch handle
const handle = client.prefetch('/posts/1');
log(`Prefetch completed: ${handle.completed}`);
await new Promise(r => setTimeout(r, 200));
log(`Prefetch completed after wait: ${handle.completed}`, 'success');

Retry
Exponential backoff with jitter, selective retry conditions, and RetryError inspection.
JavaScript 
const { HTTPBuilder } = Reixo;

const client = new HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withRetry({
    maxAttempts: 3,
    initialDelayMs: 200,
    backoffFactor: 2,
    jitter: true,
    // Only retry on 5xx and network errors (not 4xx)
    retryCondition: (err) => {
      if (err?.status >= 400 && err?.status < 500) return false;
      return true;
    }
  })
  .build();

// This request succeeds
const res = await client.get('/users/1');
log(`✓ ${res.data.name}`, 'success');

// 404 is NOT retried (4xx)
try {
  await client.get('/users/9999');
} catch (err) {
  log(`404 not retried: ${err.status}`, 'warn');
}

// withRetry() standalone utility
const { withRetry } = Reixo;
let attempts = 0;
const result = await withRetry(
  async () => {
    attempts++;
    if (attempts < 3) throw new Error('transient');
    return 'success';
  },
  { maxAttempts: 5, initialDelayMs: 50 }
);
log(`withRetry succeeded after ${attempts} attempts: ${result.value}`, 'success');

Circuit Breaker
Fail-fast with automatic CLOSED → OPEN → HALF_OPEN → CLOSED state machine.
JavaScript 
const { CircuitBreaker, CircuitState, CircuitOpenError } = Reixo;

const cb = new CircuitBreaker({
  failureThreshold: 3,   // Open after 3 consecutive failures
  resetTimeoutMs: 1000,   // Try HALF_OPEN after 1s
  successThreshold: 1,   // Close again after 1 success
});

log(`Initial state: ${cb.state}`);

// Simulate failures to trip the breaker
for (let i = 0; i < 3; i++) {
  await cb.execute(async () => { throw new Error('service down'); })
    .catch(() => {});
  log(`After failure ${i+1}: ${cb.state}`);
}

// Breaker is now OPEN — requests fail immediately
try {
  await cb.execute(async () => 'would succeed');
} catch (err) {
  if (err instanceof CircuitOpenError) {
    log(`CircuitOpenError: ${err.message}`, 'error');
  }
}

// Wait for reset window, then recover
await new Promise(r => setTimeout(r, 1100));
const result = await cb.execute(async () => 'recovered!');
log(`Recovered: "${result}", state: ${cb.state}`, 'success');

Response Caching
cache-first, stale-while-revalidate, cache metadata, and manual invalidation.
JavaScript 
const client = new Reixo.HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withCache({ ttl: 30000, strategy: 'cache-first' })
  .build();

// First request — goes to network
const r1 = await client.get('/posts/1');
log(`First : "${r1.data.title.slice(0,40)}…"`);
log(`  Cache hit: ${r1.cacheMetadata?.hit ?? false}`, 'dim');

// Second request — served from cache
const r2 = await client.get('/posts/1');
log(`Second: cache hit = ${r2.cacheMetadata?.hit}`, 'success');
if (r2.cacheMetadata) {
  log(`  age=${r2.cacheMetadata.age}s, ttl=${r2.cacheMetadata.ttl}s`, 'dim');
}

// Invalidate cache and force re-fetch
client.invalidateCache('/posts/1');
const r3 = await client.get('/posts/1');
log(`After invalidate: hit = ${r3.cacheMetadata?.hit ?? false}`);

// Stale-while-revalidate strategy
const swrClient = new Reixo.HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withCache({ ttl: 5000, strategy: 'stale-while-revalidate' })
  .build();
await swrClient.get('/posts/2'); // prime cache
const swr = await swrClient.get('/posts/2');
log(`SWR strategy: ${swr.cacheMetadata?.strategy}`, 'info');

Polling
Poll until a condition is met, with adaptive intervals and timeout protection.
JavaScript 
const { poll } = Reixo;

// Simulate a job that finishes after 4 ticks
let tick = 0;
async function checkJob() {
  tick++;
  const progress = Math.min(tick * 25, 100);
  log(`  tick ${tick}: progress ${progress}%`, 'dim');
  return { status: progress >= 100 ? 'done' : 'running', progress };
}

const { promise, cancel } = poll(checkJob, {
  interval: 300,
  until: (r) => r.status === 'done',

  // Slow polling early, fast when nearly complete
  adaptiveInterval: (r) => r.progress < 75 ? 400 : 100,

  timeout: 10000,
});

const result = await promise;
log(`Job finished: ${result.status} after ${tick} ticks`, 'success');

// poll() with maxAttempts limit
tick = 0;
const { promise: p2 } = poll(checkJob, {
  interval: 200,
  maxAttempts: 2,  // will fail — not enough attempts
});
await p2.catch(err => log(`maxAttempts: ${err.message}`, 'warn'));

MockAdapter
Test your code without a real server — fixed replies, callback handlers, delays, and error simulation.
JavaScript 
const { HTTPBuilder, MockAdapter, HTTPError, NetworkError } = Reixo;

const mock = new MockAdapter();
const client = new HTTPBuilder('https://api.example.com')
  .withTransport(mock.transport)
  .withRetry(false)
  .build();

// Fixed reply
mock.onGet('/users/1').reply(200, { id: 1, name: 'Alice' });
const user = await client.get('/users/1');
log(`User: ${user.data.name}`, 'success');

// Callback handler (conditional)
mock.onPost('/users').reply(config => {
  const body = typeof config.body === 'string'
    ? JSON.parse(config.body) : config.body;
  return body?.role === 'admin'
    ? [403, { error: 'Forbidden' }]
    : [201, { id: 42, ...body }];
});

try { await client.post('/users', { name: 'Eve', role: 'admin' }); }
catch (e) { log(`Admin blocked: ${e.status}`, 'warn'); }

const ok = await client.post('/users', { name: 'Bob', role: 'user' });
log(`User created: id=${ok.data.id}`, 'success');

// Network error
mock.onGet('/offline').networkError();
try { await client.get('/offline'); }
catch (e) { log(`NetworkError: ${e instanceof NetworkError}`, 'error'); }

log(`Total requests: ${mock.getHistory().length}`, 'info');

Logging
Text and JSON formats, log level filtering, custom prefix, and header redaction.
JavaScript 
const { ConsoleLogger, LogLevel } = Reixo;

// Text logger (default) — check browser DevTools console
const textLogger = new ConsoleLogger({
  level: LogLevel.DEBUG,
  prefix: '[MyApp:HTTP]',
  redactHeaders: ['Authorization', 'Cookie'],
});

textLogger.debug('Cache miss', { url: '/api/users' });
textLogger.info('Request sent', { method: 'GET', url: '/api/users' });
textLogger.warn('Slow response', { ms: 2400 });
log('Text format logged to DevTools console', 'info');

// JSON logger — structured for log aggregators
const jsonLogger = new ConsoleLogger({
  level: LogLevel.WARN,
  format: 'json',
  prefix: 'my-service',
  redactHeaders: ['Authorization'],
});

jsonLogger.debug('suppressed (below WARN)');
jsonLogger.info('suppressed (below WARN)');
jsonLogger.warn('High error rate', { rate: 0.12 });
log('JSON format: {timestamp,level,message,meta} in DevTools', 'info');

// Header redaction demo
const requestMeta = {
  headers: {
    authorization: 'Bearer super-secret-token',
    'content-type': 'application/json',
  }
};
textLogger.info('Outgoing request (auth redacted)', requestMeta);
log('Check DevTools — authorization header should be [REDACTED]', 'success');

Metrics
Track latency, error rates, and throughput with built-in MetricsCollector.
JavaScript 
const client = new Reixo.HTTPBuilder('https://jsonplaceholder.typicode.com')
  .withMetrics(true)
  .withRetry(false)
  .build();

// Make 6 requests (1 will 404)
await Promise.allSettled([
  client.get('/users/1'),
  client.get('/users/2'),
  client.get('/posts/1'),
  client.get('/posts/2'),
  client.get('/todos/1'),
  client.get('/users/9999'),  // 404
]);

const snap = client.metrics.getSnapshot();
log(`Requests : ${snap.requestCount}`);
log(`Errors   : ${snap.errorCount}`, snap.errorCount > 0 ? 'warn' : 'success');
log(`Avg ms   : ${snap.averageLatency.toFixed(1)}`);
log(`Min ms   : ${snap.minLatency}`, 'info');
log(`Max ms   : ${snap.maxLatency}`, 'info');

const recent = client.metrics.getRecentRequests();
log('\nPer-request breakdown:', 'dim');
recent.forEach(r => {
  const icon = r.success ? '✓' : '✗';
  log(`  ${icon} ${r.method} ${r.url} → ${r.status} (${r.duration}ms)`,
    r.success ? 'success' : 'error');
});
